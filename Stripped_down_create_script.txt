CREATE TABLE IF NOT EXISTS public.games
(
    game_id integer NOT NULL DEFAULT nextval('"Games_game_id_seq"'::regclass),
    regular_season boolean,
    away_team_id integer NOT NULL DEFAULT nextval('"Games_away_team_id_seq"'::regclass),
    home_team_id integer NOT NULL DEFAULT nextval('"Games_home_team_id_seq"'::regclass),
    week integer NOT NULL,
    game_date date,
    start_time time without time zone,
    stadium text COLLATE pg_catalog."default",
    roof boolean,
    surface text COLLATE pg_catalog."default",
    weather text COLLATE pg_catalog."default",
    wind integer,
    season integer,
    spread_line numeric(4,1),
    total_line numeric(4,1),
    division_game boolean,
    neutral_location boolean,
    home_coach text COLLATE pg_catalog."default",
    away_coach text COLLATE pg_catalog."default",
    game_id_import text COLLATE pg_catalog."default",
    CONSTRAINT "Games_pkey" PRIMARY KEY (game_id),
    CONSTRAINT "Games_away_team_id_fkey" FOREIGN KEY (away_team_id)
        REFERENCES public.teams (team_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT "Games_away_team_id_fkey1" FOREIGN KEY (away_team_id)
        REFERENCES public.teams (team_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT "Games_home_team_id_fkey" FOREIGN KEY (home_team_id)
        REFERENCES public.teams (team_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT "Games_home_team_id_fkey1" FOREIGN KEY (home_team_id)
        REFERENCES public.teams (team_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT "Games_surface_check" CHECK (surface = ANY (ARRAY['a_turf'::text, 'grass'::text, 'sportturf'::text, 'fieldturf'::text, 'matrixturf'::text, 'astroturf'::text])) NOT VALID
)
